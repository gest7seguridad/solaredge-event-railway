# Directivas nginx para Plesk - Additional nginx directives
# Copia este contenido en el campo "Additional nginx directives" de Plesk

# Configuraci칩n para servir el frontend React
location / {
    root /var/www/vhosts/gestsiete.es/solarland.gestsiete.es/backend/public;
    try_files $uri $uri/ /index.html;
    
    # Headers de seguridad
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
}

# Configuraci칩n para archivos est치ticos
location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
    root /var/www/vhosts/gestsiete.es/solarland.gestsiete.es/backend/public;
    expires 1y;
    add_header Cache-Control "public, immutable";
}

# Configuraci칩n para el API del backend
location /api {
    proxy_pass http://127.0.0.1:3000;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection 'upgrade';
    proxy_set_header Host $host;
    proxy_cache_bypass $http_upgrade;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
}

# Health check endpoint
location /api/health {
    proxy_pass http://127.0.0.1:3000/api/health;
    proxy_http_version 1.1;
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
}

# Deshabilitar acceso a archivos sensibles
location ~ /\. {
    deny all;
    return 404;
}

location ~ \.(env|git|gitignore)$ {
    deny all;
    return 404;
}
